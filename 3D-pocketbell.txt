// ============================================
// SIGN-X CORE ENGINE v4.7 (双方向コンパイル＆脳直パルスデコーダー)
// ============================================
const SIGNXCore = (() => {
  const coreState = { beingDom: null, beingDepth: null, stateLayers: [], field: null, noun: null, forceGraph: [], timeline: null };
  const REG = {
    beings: ['⚙','∞','◇','♢','♂','♀'],
    states: ['☀️','⛅','☁️','⛈️','🌙','❄️'],
    fields: ['🏠','🏢','☕','🏫','🏥','🌲','🛒','✈️','🌐','🛡️','🔥','🕳️','♾️'],
    verbs: ['V','S','E','F','G','T','I','D','M','C','P'], // v1.1完全復元：C=Connect, P=Purge
    connectors: ['→','!>','⇋','*~','✴'],
    timelines: ['.P','.N','.F']
  };

  function reset() {
    coreState.beingDom = null; coreState.beingDepth = null; coreState.stateLayers = []; coreState.field = null; coreState.noun = null; coreState.forceGraph = []; coreState.timeline = null;
  }

  // 生のLITE暗号、ブラケット完成パケットのどちらも完全パースする双方向エンジン
  function parse(code) {
    reset();

    // ① 完成版パケットのブラケット分解ルート
    const bMatch = code.match(/\[Being:\s*([^\]]+)\]/);
    if(bMatch) {
      const bContent = bMatch[1];
      if(bContent.includes('⚙')) coreState.beingDom = '⚙';
      else if(bContent.includes('∞')) coreState.beingDom = '∞';
      else if(bContent.includes('◇')) coreState.beingDom = '◇';
      else if(bContent.includes('♢')) coreState.beingDom = '♢';
      const d = bContent.match(/[+-][ⅠⅡⅢⅣ]/); if(d) coreState.beingDepth = d[0];
    } else {
      parseBeingRaw(code);
    }

    const sMatch = code.match(/\[State:\s*([^\]]+)\]/);
    if(sMatch) {
      coreState.stateLayers = sMatch[1].split(' | ');
    } else {
      parseStateRaw(code);
    }

    const fMatch = code.match(/\[Field:\s*([^\]]+)\]/);
    if(fMatch) { coreState.field = fMatch[1].trim(); } else { parseFieldRaw(code); }

    const nMatch = code.match(/⟨(.*?)⟩/);
    if(nMatch) coreState.noun = nMatch[1];

    const forceMatch = code.match(/\[Force:\s*([^\]]+)\]/);
    if(forceMatch) {
      const fStr = forceMatch[1];
      const tMatch = fStr.match(/\.(P|N|F)/); if(tMatch) coreState.timeline = tMatch[0];
      parseForceGraphRaw(fStr.replace(/\.(P|N|F)/, ''));
    } else {
      const t = code.match(/\.(P|N|F)/); if(t) coreState.timeline = t[0];
      parseForceGraphRaw(code);
    }

    validate();
    return build();
  }

  function parseBeingRaw(code) {
    if(code.includes('⚙')) coreState.beingDom = '⚙';
    else if(code.includes('∞')) coreState.beingDom = '∞';
    else if(code.includes('◇')) coreState.beingDom = '◇';
    else if(code.includes('♢') || code.includes('♂') || code.includes('♀')) coreState.beingDom = '♢';
    const depth = code.match(/[+-][ⅠⅡⅢⅣ]/); if(depth) coreState.beingDepth = depth[0];
  }

  function parseStateRaw(code) {
    const stateRegex = /([☀️⛅☁️⛈️🌙❄️][️️]*[→~]*[☀️⛅☁️⛈️🌙❄️]*[️️]*[ⅠⅡⅢ]*)/gu;
    const matches = [...code.matchAll(stateRegex)];
    matches.forEach(m => { if(m[0] && (!coreState.beingDepth || !coreState.beingDepth.includes(m[0]))) { coreState.stateLayers.push(m[0]); } });
  }

  function parseFieldRaw(code) {
    for(const f of REG.fields) { if(code.includes(f)) { coreState.field = f; break; } }
  }

  function parseForceGraphRaw(code) {
    let cleaned = code;
    // v4.2準拠 マスククレンジングによるギリシャ数字誤爆衝突の100%永久パージ
    coreState.stateLayers.forEach(layer => { cleaned = cleaned.replace(layer, ''); });
    cleaned = cleaned.replace(/\[.*?\]/g,''); cleaned = cleaned.replace(/⟨.*?⟩/g,'');
    if(coreState.beingDepth) cleaned = cleaned.replace(coreState.beingDepth, '');
    cleaned = cleaned.replace(/[☀️⛅☁️⛈️🌙❄️]/gu,''); cleaned = cleaned.replace(/\.(P|N|F)/g,''); cleaned = cleaned.replace(/[~→]/g,'');

    const tokenRegex = /(V|S|E|F|G|T|I|D|M|C|P|→|!>|⇋|\*~|✴)/g;
    const tokens = cleaned.match(tokenRegex); if(!tokens) return;
    for(let i=0; i<tokens.length; i++) {
      const t = tokens[i];
      if(REG.verbs.includes(t)) {
        const next = tokens[i+1]; const connector = REG.connectors.includes(next) ? next : null;
        coreState.forceGraph.push({ verb: t, connector });
      }
    }
  }

  function validate() { if(coreState.forceGraph.length > 0 && !coreState.beingDom) { console.warn('SIGN-X WARNING :: FORCE without BEING'); } }

  document.getElementById('inputText').value = '';
  function build() {
    let out = '';
    if(coreState.beingDom || coreState.beingDepth) { out += `[Being: ${coreState.beingDom || ''}${coreState.beingDepth || ''}] `; }
    if(coreState.stateLayers.length > 0) { out += `[State: ${coreState.stateLayers.join(' | ')}] `; }
    if(coreState.field) { out += `[Field: ${coreState.field}] `; }
    if(coreState.noun) { out += `⟨${coreState.noun}⟩ `; }
    if(coreState.forceGraph.length > 0) {
      out += '[Force: ';
      coreState.forceGraph.forEach(node => { out += node.verb; if(node.connector) out += node.connector; });
      if(coreState.timeline) out += coreState.timeline; out += ']';
    }
    return out.trim();
  }

  function semanticExpand() {
    const mapV = { V:'verify', S:'speak', E:'evaluateEmotion', F:'find', G:'generate', T:'think', I:'intend', D:'deploy', M:'modify', C:'connect', P:'purge' };
    const mapC = { '→':'causal', '!>':'forced', '⇋':'bidirectional', '*~':'resonance', '✴':'breakthrough' };
    const mapT = { '.P':'past', '.N':'now', '.F':'future' };
    return {
      being: coreState.beingDom, depth: coreState.beingDepth, states: coreState.stateLayers, field: coreState.field, noun: coreState.noun,
      actions: coreState.forceGraph.map(f => ({ action: mapV[f.verb] || f.verb, transition: mapC[f.connector] || null })), timeline: mapT[coreState.timeline] || null
    };
  }

  return { parse, build, semanticExpand, state: coreState };
})();

  if(/自宅|家/.test(text)) code += '🏠'; else if(/カフェ|喫茶店/.test(text)) code += '☕'; else if(/会社|オフィス/.test(text)) code += '🏢';

}

  document.getElementById('dictKey').value = ''; document.getElementById('dictVal').value = ''; showToast('辞書ロック 📮');
}
